Usage Example

<ava-nav-bar [containerStyles]="navBarStyles">
  <aava-button 
    label="Dashboard" 
    variant="primary" 
    [pill]="true"
    iconName="layout-dashboard"
    iconPosition="left"
    [customStyles]="defaultButtonStyles"
    class="nav-item"
  ></aava-button>
  
  <aava-button 
    label="Libraries" 
    variant="primary" 
    [pill]="true"
    iconName="book"
    iconPosition="left"
    [customStyles]="activeButtonStyles"
    class="nav-item nav-item--active"
  ></aava-button>
  
  <!-- Can even use custom elements -->
  <a href="/about" class="custom-nav-link">About</a>
</ava-nav-bar>

Container Styling

// Container styling interface
interface NavBarContainerStyles {
  background?: string;
  padding?: string;
  borderRadius?: string;
  boxShadow?: string;
  border?: string;
  gap?: string;
  outerGlow?: boolean;
  outerGlowColor?: string;
  outerGlowIntensity?: string;
  width?: string;
  height?: string;
  backdropFilter?: string;
}

// Example usage
const navBarStyles: NavBarContainerStyles = {
  background: 'rgba(255, 255, 255, 0.95)',
  borderRadius: '50px',
  padding: '8px',
  boxShadow: '0 8px 32px rgba(0, 0, 0, 0.1)',
  backdropFilter: 'blur(10px)',
  border: '1px solid rgba(255, 255, 255, 0.2)',
  gap: '8px',
  outerGlow: true,
  outerGlowColor: 'rgba(59, 130, 246, 0.3)',
  outerGlowIntensity: '20px'
};

Features

  • ✅ Content projection for ultimate flexibility
  • ✅ Use any component (aava-button, custom buttons, links, etc.)
  • ✅ Direct control over each nav item's styling
  • ✅ Container styling via containerStyles prop
  • ✅ Responsive design for mobile devices
  • ✅ Theme-aware styling with CSS variables
  • ✅ Clean separation of concerns
  • ✅ No complex prop configurations
  • ✅ Easy to customize and extend
  • ✅ Composable with any content
